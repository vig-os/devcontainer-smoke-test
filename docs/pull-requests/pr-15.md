import re

try:
    import tomllib
except ImportError:  # pragma: no cover - Python < 3.11
    tomllib = None

def load_blocklist(path: Path) -> list[str]:
    """Load blocklisted patterns from a TOML file.

    Supports either:
      - a top-level `patterns = ["regex1", "regex2", ...]` list, or
      - `[[fingerprints]]` tables each with a `pattern = "..."` field.
    """
    if tomllib is None:
        # Environment does not support TOML parsing via tomllib; treat as no blocklist.
        return []

    with path.open("rb") as f:
        data = tomllib.load(f)

    patterns: list[str] = []

    top_level_patterns = data.get("patterns")
    if isinstance(top_level_patterns, list):
        for p in top_level_patterns:
            if isinstance(p, str):
                patterns.append(p)

    fingerprints = data.get("fingerprints")
    if isinstance(fingerprints, list):
        for item in fingerprints:
            if isinstance(item, dict):
                pat = item.get("pattern")
                if isinstance(pat, str):
                    patterns.append(pat)

    return patterns


def contains_agent_fingerprint(content: str, patterns: list[str]) -> str | None:
    """Return the first matching pattern if any fingerprint is found in content."""
    for pattern in patterns:
        try:
            if re.search(pattern, content, flags=re.IGNORECASE):
                return pattern
        except re.error:
            # Skip invalid regex patterns rather than failing the entire check.
            continue
    return None


def main() -> int:
